import { useEffect, useMemo, useState } from "react";
import { Link, useLocation, useParams } from "react-router-dom";
import { useTheme } from "@/components/common/ThemeContext";
import api from "@/api/api";
import DOMPurify from "dompurify";
import {
    ArrowLeft, CheckCircle2, XCircle, HelpCircle, Clock, Loader2, Download, FileText, User2
} from "lucide-react";

type Task = {
    id: number;
    number?: number | null;
    name: string;
    description?: string | null;
    allow_resubmitting_task: boolean;
    // NOTE: в бэке нет max_score у Task — ниже используем fallback 100
    // max_score?: number;
};

type MiniCourse = { id: number; name: string };
type MiniUser   = { id: number; first_name: string; last_name: string; email: string; profile?: { photo?: string | null } | null };

type AnswerFile = { id: number; file_name: string; file_url: string; file_size: number };
type Grade = {
    score: number | null;
    max_score: number;
    percentage: number;
    letter_grade: string;
    graded_by?: { first_name: string; last_name: string };
    feedback_text?: string;
} | null;

type Answer = {
    id: number;
    description: string;
    status: "in_review" | "approved" | "have_flaws" | "rejected";
    files: AnswerFile[];
    grade: Grade;
    updated_at: string;
    task: { id: number; course: MiniCourse };
    user: MiniUser;
};

type TeacherReviewPayload = { answers: Answer[] };

const cx = (...c: (string | false | null | undefined)[]) => c.filter(Boolean).join(" ");

export default function StudentReview() {
    const { studentId, courseId } = useParams<{ studentId: string; courseId: string }>();
    const location = useLocation() as any;
    const passedStudent: MiniUser | undefined = location.state?.student;
    const passedCourse : MiniCourse | undefined = location.state?.course;

    const { actualTheme } = useTheme();
    const isDark = actualTheme === "dark";

    const [loading, setLoading] = useState(true);
    const [tasks, setTasks] = useState<Task[]>([]);
    const [answers, setAnswers] = useState<Answer[]>([]);
    const [selectedTaskId, setSelectedTaskId] = useState<number | null>(null);
    const [submitting, setSubmitting] = useState<"approve"|"flaws"|"reject"|null>(null);

    // fetch tasks & answers of this course
    useEffect(() => {
        let on = true;
        (async () => {
            try {
                const [t, a] = await Promise.all([
                    api.get<Task[]>(`course/courses/${courseId}/tasks/`),
                    api.get<TeacherReviewPayload>(`answers/teacher-review/?course_id=${courseId}`)
                ]);
                if (!on) return;
                const sorted = (t.data || []).slice().sort((a,b) => (a.number||0) - (b.number||0));
                setTasks(sorted);
                const onlyThisStudent = (a.data.answers || []).filter(x => String(x.user.id) === String(studentId));
                setAnswers(onlyThisStudent);
                setSelectedTaskId(prev => prev ?? (sorted[0]?.id ?? null));
            } catch (e) {
                console.error(e);
            } finally {
                if (on) setLoading(false);
            }
        })();
        return () => { on = false; };
    }, [studentId, courseId]);

    const answersByTask = useMemo(() => {
        const mp = new Map<number, Answer>();
        answers.forEach(a => mp.set(a.task.id, a));
        return mp;
    }, [answers]);

    const selectedTask = tasks.find(t => t.id === selectedTaskId) || null;
    const selectedAnswer = selectedTask ? answersByTask.get(selectedTask.id) || null : null;

    const statusIcon = (status?: Answer["status"]) => {
        switch (status) {
            case "approved":   return <CheckCircle2 className="h-5 w-5 text-emerald-500" />;
            case "have_flaws": return <HelpCircle   className="h-5 w-5 text-amber-500" />;
            case "rejected":   return <XCircle      className="h-5 w-5 text-rose-500" />;
            case "in_review":  return <Clock        className="h-5 w-5 opacity-70" />;
            default:           return <FileText     className="h-5 w-5 opacity-50" />; // нет ответа
        }
    };

    const student: MiniUser | null = passedStudent ?? (selectedAnswer?.user ?? null);
    const course : MiniCourse | null = passedCourse  ?? (selectedAnswer?.task.course ?? null);

    async function review(action: "approve" | "flaws" | "reject", payload: { score?: number | null; feedback_text?: string }) {
        if (!selectedAnswer) return;
        const answerId = selectedAnswer.id;
        setSubmitting(action);
        try {
            const statusMap = {
                approve: "approved",
                flaws:   "have_flaws",
                reject:  "rejected",
            } as const;

            // из grade берём max_score (если нет — 100 как в модели Grade)
            const maxScore = selectedAnswer.grade?.max_score ?? 100;

            const body: any = {
                status: statusMap[action],
                max_score: maxScore, // требование: «макс балл из Task», но в модели Task его нет — fallback 100
            };
            if (typeof payload.score !== "undefined") body.score = payload.score;
            if (typeof payload.feedback_text !== "undefined") body.feedback_text = (payload.feedback_text || "").trim();

            const res = await api.patch(`answers/${answerId}/check/`, body);
            // обновить в локальном стейте
            setAnswers(prev => prev.map(a => a.id === answerId ? { ...a, ...res.data } as any : a));
        } catch (e) {
            console.error(e);
            alert("Не удалось сохранить решение. Проверьте соединение и права.");
        } finally {
            setSubmitting(null);
        }
    }

    if (loading) {
        return <div className={cx("p-8", isDark ? "text-slate-400" : "text-neutral-600")}>Загрузка…</div>;
    }

    return (
        <div className={cx("mx-auto max-w-screen-2xl px-4 py-6", isDark ? "text-slate-200" : "text-neutral-900")}>
            {/* header */}
            <div className="mb-4 flex items-center gap-3">
                <Link
                    to="/teacher/students"
                    className={cx(
                        "inline-flex items-center gap-2 rounded-xl border px-3 py-2 text-sm hover:border-neutral-300 hover:bg-neutral-50",
                        isDark ? "bg-slate-800 border-slate-700 text-slate-300 hover:bg-slate-700" : "bg-white border-neutral-200"
                    )}
                >
                    <ArrowLeft className="h-4 w-4" /> Назад к студентам
                </Link>
                <span className={cx("text-sm", isDark ? "text-slate-400" : "text-neutral-500")}>/</span>
                <div className="flex items-center gap-2">
                    <Avatar url={student?.profile?.photo} />
                    <div className="text-sm">
                        <div className="font-medium">{student ? `${student.first_name} ${student.last_name}` : "Студент"}</div>
                        <div className={cx("text-xs", isDark ? "text-slate-400" : "text-neutral-500")}>
                            {student?.email || "—"} {course ? `• ${course.name}` : ""}
                        </div>
                    </div>
                </div>
            </div>

            <div className="grid gap-4 md:grid-cols-[320px,1fr]">
                {/* sidebar: tasks */}
                <aside className={cx(
                    "rounded-2xl border overflow-hidden",
                    isDark ? "bg-slate-800 border-slate-700" : "bg-white border-neutral-200"
                )}>
                    <div className={cx(
                        "px-4 py-3 text-sm font-medium",
                        isDark ? "bg-slate-800 text-slate-300" : "bg-neutral-50 text-neutral-700"
                    )}>Задания</div>
                    <ul className="divide-y">
                        {tasks.map(t => {
                            const ans = answersByTask.get(t.id) || null;
                            return (
                                <li
                                    key={t.id}
                                    onClick={() => setSelectedTaskId(t.id)}
                                    className={cx(
                                        "flex items-center gap-3 px-4 py-3 cursor-pointer",
                                        selectedTaskId === t.id ? (isDark ? "bg-slate-700/50" : "bg-neutral-50") : "hover:bg-black/5",
                                        isDark && "hover:bg-white/5"
                                    )}
                                >
                                    <div className="shrink-0">{statusIcon(ans?.status)}</div>
                                    <div className="min-w-0">
                                        <div className="truncate text-sm font-medium">{t.number ? `${t.number}. `: ""}{t.name}</div>
                                        <div className={cx("text-xs truncate", isDark ? "text-slate-400" : "text-neutral-500")}>
                                            {ans ? prettyStatus(ans.status) : "Ответа нет"}
                                        </div>
                                    </div>
                                </li>
                            );
                        })}
                    </ul>
                </aside>

                {/* main */}
                <main className="grid gap-4">
                    {/* Task description */}
                    <div className={cx(
                        "rounded-2xl border p-5",
                        isDark ? "bg-slate-800 border-slate-700" : "bg-white border-neutral-200"
                    )}>
                        <div className="flex items-start justify-between gap-3">
                            <h2 className="text-xl font-semibold">
                                {selectedTask ? (selectedTask.number ? `${selectedTask.number}. ` : "") + selectedTask.name : "Задание"}
                            </h2>
                            {selectedTask && (
                                <span className={cx(
                                    "text-xs rounded-full px-2 py-1",
                                    selectedTask.allow_resubmitting_task
                                        ? (isDark ? "bg-emerald-900/30 text-emerald-300" : "bg-emerald-100 text-emerald-700")
                                        : (isDark ? "bg-slate-700 text-slate-300" : "bg-neutral-100 text-neutral-700")
                                )}>
                  {selectedTask.allow_resubmitting_task ? "Разрешена повторная отправка" : "Только одна отправка"}
                </span>
                            )}
                        </div>
                        <div
                            className={cx(
                                "prose mt-3 max-w-none text-[0.95rem] leading-7 [&_img]:max-w-full [&_img]:rounded-lg",
                                isDark ? "prose-invert text-slate-300" : "text-neutral-800"
                            )}
                            dangerouslySetInnerHTML={{ __html: DOMPurify.sanitize(selectedTask?.description || "—") }}
                        />
                    </div>

                    {/* Student answer */}
                    <div className={cx(
                        "rounded-2xl border p-5",
                        isDark ? "bg-slate-800 border-slate-700" : "bg-white border-neutral-200"
                    )}>
                        <h3 className="text-lg font-semibold">Ответ студента</h3>

                        {!selectedAnswer && (
                            <div className={cx("mt-3 text-sm", isDark ? "text-slate-400" : "text-neutral-600")}>
                                Ответ пока не отправлен.
                            </div>
                        )}

                        {selectedAnswer && (
                            <>
                                <div className={cx("mt-2 text-sm", isDark ? "text-slate-300" : "text-neutral-800")}>
                                    {selectedAnswer.description || "—"}
                                </div>

                                {/* files */}
                                {selectedAnswer.files?.length ? (
                                    <div className="mt-3">
                                        <div className="text-sm font-medium mb-1">Файлы:</div>
                                        <ul className="space-y-1">
                                            {selectedAnswer.files.map(f => (
                                                <li key={f.id} className="flex items-center gap-2 text-sm">
                                                    <a className={cx("underline", isDark ? "text-blue-400" : "text-blue-600")}
                                                       href={f.file_url} target="_blank" rel="noreferrer">
                                                        {f.file_name}
                                                    </a>
                                                    <span className={cx("text-xs", isDark ? "text-slate-400" : "text-neutral-500")}>
                            ({Math.round((f.file_size || 0) / 1024)} KB)
                          </span>
                                                </li>
                                            ))}
                                        </ul>
                                    </div>
                                ) : null}

                                {/* grade summary */}
                                <div className="mt-4 flex flex-wrap items-center gap-2 text-sm">
                  <span className={cx("inline-flex items-center gap-2 px-2 py-1 rounded-full",
                      badgeByStatus(selectedAnswer.status, isDark)
                  )}>
                    {statusIcon(selectedAnswer.status)}
                      {prettyStatus(selectedAnswer.status)}
                  </span>

                                    {selectedAnswer.grade && (
                                        <>
                                            <span className="opacity-60">•</span>
                                            <span>
                        Балл: {selectedAnswer.grade.score ?? "—"}/{selectedAnswer.grade.max_score}
                                                {typeof selectedAnswer.grade.percentage === "number" ? ` (${Math.round(selectedAnswer.grade.percentage)}%)` : ""}
                      </span>
                                            <span className="opacity-60">•</span>
                                            <span className={cx("inline-flex items-center rounded px-1.5 py-0.5",
                                                letterShade(selectedAnswer.grade.letter_grade, isDark)
                                            )}>
                        {selectedAnswer.grade.letter_grade}
                      </span>
                                            {selectedAnswer.grade.feedback_text ? (
                                                <>
                                                    <span className="opacity-60">•</span>
                                                    <span>Комментарий: {selectedAnswer.grade.feedback_text}</span>
                                                </>
                                            ) : null}
                                        </>
                                    )}
                                </div>

                                {/* actions */}
                                <ReviewActions
                                    disabled={!!submitting}
                                    defaultScore={selectedAnswer.grade?.score ?? null}
                                    maxScore={selectedAnswer.grade?.max_score ?? 100}
                                    onApprove={(score, feedback) => review("approve", { score, feedback_text: feedback })}
                                    onNeedFix={(feedback, score) => review("flaws",   { score: score ?? selectedAnswer.grade?.score ?? null, feedback_text: feedback })}
                                    onReject={(feedback)     => review("reject",  { feedback_text: feedback })}
                                    isDark={isDark}
                                    submitting={submitting}
                                />
                            </>
                        )}
                    </div>
                </main>
            </div>
        </div>
    );
}

function prettyStatus(s: Answer["status"]) {
    switch (s) {
        case "approved": return "Принято";
        case "have_flaws": return "Есть замечания";
        case "rejected": return "Отклонено";
        case "in_review": return "На проверке";
        default: return "—";
    }
}

function badgeByStatus(s: Answer["status"], isDark: boolean) {
    switch (s) {
        case "approved":   return isDark ? "bg-emerald-900/30 text-emerald-300" : "bg-emerald-100 text-emerald-700";
        case "have_flaws": return isDark ? "bg-amber-900/30 text-amber-300" : "bg-amber-100 text-amber-700";
        case "rejected":   return isDark ? "bg-rose-900/30 text-rose-300" : "bg-rose-100 text-rose-700";
        case "in_review":  return isDark ? "bg-slate-700 text-slate-300" : "bg-neutral-200 text-neutral-800";
        default:           return isDark ? "bg-slate-700 text-slate-300" : "bg-neutral-200 text-neutral-800";
    }
}
function letterShade(l: string, isDark: boolean) {
    switch (l) {
        case "A": return isDark ? "bg-green-900/30 text-green-300"  : "bg-green-100 text-green-700";
        case "B": return isDark ? "bg-blue-900/30 text-blue-300"    : "bg-blue-100 text-blue-700";
        case "C": return isDark ? "bg-amber-900/30 text-amber-300"  : "bg-amber-100 text-amber-700";
        case "D": return isDark ? "bg-orange-900/30 text-orange-300": "bg-orange-100 text-orange-700";
        default:  return isDark ? "bg-rose-900/30 text-rose-300"    : "bg-rose-100 text-rose-700";
    }
}

function prettySize(b: number) {
    if (!b) return "0 B";
    if (b < 1024) return `${b} B`;
    if (b < 1024*1024) return `${(b/1024).toFixed(1)} KB`;
    return `${(b/1024/1024).toFixed(1)} MB`;
}

function Avatar({ url }: { url?: string | null }) {
    const { actualTheme } = useTheme();
    const isDark = actualTheme === "dark";
    if (!url) {
        return (
            <div className={cx(
                "flex h-9 w-9 items-center justify-center rounded-full",
                isDark ? "bg-slate-700 text-slate-300" : "bg-neutral-200 text-neutral-700"
            )}>
                <User2 className="h-4 w-4" />
            </div>
        );
    }
    return <img src={url} className="h-9 w-9 rounded-full object-cover" />;
}

function ReviewActions({
                           defaultScore,
                           maxScore,
                           onApprove,
                           onNeedFix,
                           onReject,
                           disabled,
                           submitting,
                           isDark
                       }: {
    defaultScore: number | null;
    maxScore: number;
    onApprove: (score: number | null, feedback?: string) => void;
    onNeedFix: (feedback: string, score?: number | null) => void;
    onReject: (feedback: string) => void;
    disabled?: boolean;
    submitting: "approve"|"flaws"|"reject"|null;
    isDark: boolean;
}) {
    const [score, setScore] = useState<number | "">(defaultScore ?? "");
    const [feedback, setFeedback] = useState("");

    const btnBase = "inline-flex items-center gap-2 rounded-xl px-3 py-2 text-sm disabled:opacity-60";

    return (
        <div className="mt-5 grid gap-3">
            {/* score + feedback */}
            <div className="grid gap-3 md:grid-cols-2">
                <div className="space-y-1">
                    <label className={cx("text-sm", isDark ? "text-slate-300" : "text-neutral-700")}>
                        Балл (0–{maxScore})
                    </label>
                    <input
                        type="number"
                        min={0}
                        max={maxScore}
                        value={score}
                        onChange={(e) => {
                            const v = e.target.value === "" ? "" : Math.max(0, Math.min(maxScore, Number(e.target.value)));
                            setScore(v as any);
                        }}
                        className={cx(
                            "w-full rounded-xl border px-3 py-2 outline-none focus:ring-2",
                            isDark ? "bg-slate-700 border-slate-600 text-slate-200 focus:ring-slate-500/30" :
                                "bg-white border-neutral-300 text-neutral-900 focus:ring-black/10"
                        )}
                        placeholder={`0…${maxScore}`}
                    />
                </div>
                <div className="space-y-1">
                    <label className={cx("text-sm", isDark ? "text-slate-300" : "text-neutral-700")}>
                        Комментарий (для студента)
                    </label>
                    <textarea
                        value={feedback}
                        onChange={(e) => setFeedback(e.target.value)}
                        className={cx(
                            "min-h-[44px] w-full rounded-xl border px-3 py-2 outline-none focus:ring-2",
                            isDark ? "bg-slate-700 border-slate-600 text-slate-200 focus:ring-slate-500/30" :
                                "bg-white border-neutral-300 text-neutral-900 focus:ring-black/10"
                        )}
                        placeholder="Коротко поясните решение"
                    />
                </div>
            </div>

            <div className="flex flex-wrap gap-2">
                <button
                    disabled={disabled}
                    onClick={() => onApprove(score === "" ? null : Number(score), feedback)}
                    className={cx(btnBase, isDark ? "bg-emerald-600 text-white hover:bg-emerald-700" : "bg-emerald-600 text-white hover:bg-emerald-700")}
                    title="Принять ответ"
                >
                    {submitting === "approve" ? <Loader2 className="h-4 w-4 animate-spin" /> : <CheckCircle2 className="h-4 w-4" />}
                    Принять
                </button>

                <button
                    disabled={disabled || !feedback.trim()}
                    onClick={() => onNeedFix(feedback.trim(), score === "" ? undefined : Number(score))}
                    className={cx(btnBase, isDark ? "bg-amber-500 text-black hover:bg-amber-400" : "bg-amber-400 text-black hover:bg-amber-300")}
                    title="Запросить уточнения — студент сможет пересдать"
                >
                    {submitting === "flaws" ? <Loader2 className="h-4 w-4 animate-spin" /> : <HelpCircle className="h-4 w-4" />}
                    Уточнить
                </button>

                <button
                    disabled={disabled || !feedback.trim()}
                    onClick={() => onReject(feedback.trim())}
                    className={cx(btnBase, isDark ? "bg-rose-600 text-white hover:bg-rose-700" : "bg-rose-600 text-white hover:bg-rose-700")}
                    title="Отклонить ответ"
                >
                    {submitting === "reject" ? <Loader2 className="h-4 w-4 animate-spin" /> : <XCircle className="h-4 w-4" />}
                    Отклонить
                </button>
            </div>

            <p className={cx("text-xs mt-1", isDark ? "text-slate-400" : "text-neutral-500")}>
                Примечание: при статусах «Есть замечания» и «Отклонено» студент может отправить решение повторно,
                даже если в задаче глобально запрещены повторные отправки.
            </p>
        </div>
    );
}

function prettyDate(s?: string | null) {
    return s ? new Date(s).toLocaleString() : "—";
}

                                                                            selectedTask.answer.grade.letter_grade === 'F' ? 'text-red-400' : 'text-blue-400'
                                                                        }`}>
                                                                            {selectedTask.answer.grade.letter_grade}
                                                                        </div>
                                                                        <p className={`text-sm font-medium ${isDark ? 'text-slate-400' : 'text-slate-600'}`}>
                                                                            Оценка
                                                                        </p>
                                                                    </div>
                                                                </div>

                                                                <div className="mb-6">
                                                                    <span className={`text-sm font-medium ${isDark ? 'text-slate-400' : 'text-slate-600'}`}>
                                                                        Комментарий:
                                                                    </span>
                                                                    <p className="mt-2 font-medium text-lg leading-relaxed">
                                                                        {selectedTask.answer.grade.feedback_text}
                                                                    </p>
                                                                </div>

                                                                <div className={`p-5 rounded-xl backdrop-blur-sm border transition-all duration-200 ${
                                                                    isDark
                                                                        ? 'bg-slate-800/50 border-slate-700/50'
                                                                        : 'bg-white/70 border-slate-200/50'
                                                                }`}>
                                                                    <div className="flex items-center gap-3 mb-3">
                                                                        <div className="p-2 rounded-lg bg-gradient-to-r from-indigo-500 to-purple-500">
                                                                            <User className="w-4 h-4 text-white" />
                                                                        </div>
                                                                        <span className="font-bold text-lg">
                                                                            {selectedTask.answer.grade.graded_by.first_name} {selectedTask.answer.grade.graded_by.last_name}
                                                                        </span>
                                                                    </div>
                                                                    <div className="flex items-center gap-6 text-sm">
                                                                        <div className="flex items-center gap-2">
                                                                            <Calendar className="w-4 h-4 text-blue-400" />
                                                                            <span className="font-medium">
                                                                                {formatDate(selectedTask.answer.grade.created_at)}
                                                                            </span>
                                                                        </div>
                                                                        <div className={`px-3 py-1 rounded-full text-xs font-medium backdrop-blur-sm ${
                                                                            isDark
                                                                                ? 'bg-indigo-500/20 text-indigo-300 border border-indigo-500/30'
                                                                                : 'bg-indigo-500/20 text-indigo-700 border border-indigo-500/30'
                                                                        }`}>
                                                                            {selectedTask.answer.grade.graded_by.role}
                                                                        </div>
                                                                    </div>
                                                                </div>
                                                            </div>
                                                        )}
                                                    </div>
                                                ) : (
                                                    <div className={`p-12 rounded-2xl text-center backdrop-blur-sm border transition-all duration-200 hover:shadow-lg ${
                                                        isDark
                                                            ? 'bg-gradient-to-br from-slate-800/40 to-slate-900/40 border-slate-700/50'
                                                            : 'bg-gradient-to-br from-white/60 to-slate-50/60 border-slate-200/50'
                                                    }`}>
                                                        <div className="relative mb-6">
                                                            <Circle className={`w-16 h-16 mx-auto ${isDark ? 'text-slate-500' : 'text-slate-400'} transition-all duration-200`} />
                                                            <div className="absolute inset-0 w-16 h-16 mx-auto rounded-full border-2 border-dashed border-slate-400/30 animate-pulse"></div>
                                                        </div>
                                                        <h3 className="font-bold text-xl mb-3 bg-gradient-to-r from-slate-400 to-slate-300 bg-clip-text text-transparent">
                                                            Ответ не отправлен
                                                        </h3>
                                                        <p className={`text-base ${isDark ? 'text-slate-400' : 'text-slate-600'} max-w-md mx-auto leading-relaxed`}>
                                                            Студент еще не отправил решение для этого задания. Как только ответ будет загружен, он появится здесь.
                                                        </p>
                                                    </div>
                                                )}
                                            </div>
                                        </CustomScrollbar>
                                    ) : (
                                        <div className="flex-1 flex items-center justify-center">
                                            <div className="text-center max-w-md mx-auto p-8">
                                                <div className="relative mb-8">
                                                    <FileText className={`w-20 h-20 mx-auto ${isDark ? 'text-slate-500' : 'text-slate-400'} transition-all duration-200`} />
                                                    <div className="absolute inset-0 w-20 h-20 mx-auto rounded-full border-2 border-dashed border-slate-400/30 animate-pulse"></div>
                                                </div>
                                                <h3 className="font-bold text-2xl mb-4 bg-gradient-to-r from-slate-400 to-slate-300 bg-clip-text text-transparent">
                                                    Выберите задание
                                                </h3>
                                                <p className={`text-base ${isDark ? 'text-slate-400' : 'text-slate-600'} leading-relaxed`}>
                                                    Выберите задание из списка слева для просмотра подробной информации о решении студента
                                                </p>
                                            </div>
                                        </div>
                                    )}
                                </div>
                            </>
                        )}
                    </div>
                </div>
            </div>
        </div>,
        document.body
    );
}
